class Solution {
    public Node connect(Node root) {
        connectRoot(root);
        return root;
    }
    public void connectRoot(Node root) {
        if(root == null)
            return;
        if(root.left != null) {
            root.left.next = root.right;
            if(root.next != null)
                root.right.next = root.next.left;
        }
        connectRoot(root.left);
        connectRoot(root.right);
    }
}
