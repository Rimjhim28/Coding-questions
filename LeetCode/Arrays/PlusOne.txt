class Solution {
    public int[] plusOne(int[] digits) {
        for(int i = digits.length-1; i >= 0; i--) {
            if(digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            digits[i] = 0;
        }
        int ans[] = new int[digits.length+1];
        for(int i = 0; i < digits.length; i++)
            ans[i+1] = digits[i];
        ans[0] = 1;
        return ans;
    }
}
