/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public ListNode removeElements(ListNode head, int val) {
        while(head != null && head.val == val) head = head.next;
        ListNode p = head;
        while(p != null && p.next != null) {
            if(p.next.val == val)
                p.next = p.next.next;
            else
                p = p.next;
        }
        return head;
    }
}
